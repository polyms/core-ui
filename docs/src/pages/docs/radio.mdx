## Usage

```tsx
import { Radio, RadioGroup } from '@polyms/core-ui'

const radioGroup = (
  <RadioGroup name='plan' onValueChange={setPlan} value={plan}>
    <Radio value='free'>Free</Radio>
    <Radio value='pro'>Pro</Radio>
  </RadioGroup>
)
```
